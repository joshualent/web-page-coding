import { Sandpack } from "@codesandbox/sandpack-react";
import { githubLight } from "@codesandbox/sandpack-themes";

export const App = () => {
  const files = {
    "/index.html": {
      code: `<!DOCTYPE html>
<html>
  <head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
  </head>
  <body>
  </body>
</html>`
    }
  }
  
  return (
    <Sandpack
      files={files} 
      theme={githubLight} 
      template="static"
    />
  )  
}
